  • 1.5 En la tabla siguiente y en la grfica de la parte 1.4 indique los rangos de precios en los que la empresa obtiene los resultados que refiere la primera columna.

import pandas as pd
import numpy as np
import openpyxl

# Definir los datos para cada columna
data = {
    'q': list(range(0, 11)),  # Valores para la columna 'q'
    'IT': [np.nan] * 11,       # Valores vacos para la columna 'IT'
    'IM': [np.nan] * 11,       # Valores vacos para la columna 'IM'
    'CFT': [175] * 11,         # Costo Fijo Total constante en todas las filas
    'CVT': [0, 100, 180, 230, 305, 405, 530, 680, 855, 1055, 1285], 
    'CT': [np.nan] * 11,       # Valores vacos para la columna 'CT'
    'CFMe': [np.nan] * 11,     # Valores vacos para la columna 'CFMe'
    'CVMe': [np.nan] * 11,     # Valores vacos para la columna 'CVMe'
    'CM': [np.nan] * 11,       # Valores vacos para la columna 'CM'
    'BM': [np.nan] * 11,       # Valores vacos para la columna 'BM'
    '': [np.nan] * 11         # Valores vacos para la columna ''
}

# Crear el DataFrame
df = pd.DataFrame(data)

# Calcular el Ingreso Total (IT)
df['IT'] = df['q']*150

# Calcular el Ingreso Marginal (IM)
df['IM'] = (df['IT'].diff())/(df['q'].diff())

# Calcular el Costo Total (CT)
df['CT'] = df['CFT'] + df['CVT']

# Calcular el Costo Fijo Medio (CFMe)
df['CFMe'] = round(df['CFT'] / df['q'], 2)

# Calcular el Costo Variable Medio (CVMe)
df['CVMe'] = round(df['CVT'] / df['q'], 2)

# Calcular el Costo Marginal (CM)
df['CM'] = (df['CT'].diff())/(df['q'].diff())

# Calcular el Beneficio Marginal (BM)
df['BM'] = df['IM'] - df['CM']

# Calcular el Beneficio ()
df[''] = df['IT'] - df['CT']

print(df)

import pandas as pd
import matplotlib.pyplot as plt
from scipy.interpolate import make_interp_spline
import numpy as np

# Cargar el DataFrame desde el archivo CSV
df = pd.read_csv('Datos_1_4.csv')

# Limpiar los datos eliminando valores NaN e infinitos
df.replace([np.inf, -np.inf], np.nan, inplace=True)
df.dropna(inplace=True)

# Crear la grfica
plt.figure(figsize=(10, 6))

# Funcin para suavizar las lneas
def smooth_line(x, y, num_points=500):
    x_smooth = np.linspace(x.min(), x.max(), num_points)
    spl = make_interp_spline(x, y, k=3)
    y_smooth = spl(x_smooth)
    return x_smooth, y_smooth

# Graficar CVMe con suavizado
x_smooth, y_smooth = smooth_line(df['q'], df['CVMe'])
plt.plot(x_smooth, y_smooth, label='CVMe', color='blue')

# Graficar CTMe con suavizado
x_smooth, y_smooth = smooth_line(df['q'], df['CTMe'])
plt.plot(x_smooth, y_smooth, label='CTMe', color='orange')

# Graficar CM con suavizado
x_smooth, y_smooth = smooth_line(df['q'], df['CM'])
plt.plot(x_smooth, y_smooth, label='CM', color='green')

# Graficar IM con suavizado
x_smooth, y_smooth = smooth_line(df['q'], df['IM'])
plt.plot(x_smooth, y_smooth, label='IM', color='black', linestyle='--')

# Etiquetas y ttulo
plt.xlabel('Nivel de Produccin (q)')
plt.ylabel('Precio')
plt.title('Curvas de CVMe, CTMe, CM e IM')
plt.legend()

# Mostrar la grfica
plt.grid(True)
plt.show()
